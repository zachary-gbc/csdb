DROP TABLE IF EXISTS Users;
CREATE TABLE Users(
  Usr_ID INT NOT NULL AUTO_INCREMENT,
  Usr_Name VARCHAR(100) NOT NULL UNIQUE,
  Usr_Email VARCHAR(100) NOT NULL UNIQUE,
  Usr_FirstName VARCHAR(100) NOT NULL,
  Usr_LastName VARCHAR(100) NOT NULL,
  Usr_Password VARCHAR(100) NOT NULL,
  Usr_PushoverCode VARCHAR(100) NULL,
  PRIMARY KEY(Usr_ID)
);

DROP TABLE IF EXISTS Variables;
CREATE TABLE Variables(
  Var_ID INT NOT NULL AUTO_INCREMENT,
  Var_System VARCHAR(50) NOT NULL,
  Var_Name VARCHAR(100) NOT NULL,
  Var_Value VARCHAR(150) NULL,
  PRIMARY KEY(Var_ID)
);

DROP TABLE IF EXISTS PushoverLog;
CREATE TABLE PushoverLog(
  PO_ID INT NOT NULL AUTO_INCREMENT,
  PO_Device VARCHAR(100) NOT NULL,
  PO_Title VARCHAR(100) NOT NULL,
  PO_Response VARCHAR(100) NULL,
  PO_AlertSent VARCHAR(100) NULL,
  PO_DateTime VARCHAR(100) NULL,
  PRIMARY KEY(PO_ID)
);
